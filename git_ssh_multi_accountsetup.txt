#Generate a New SSH Key

ssh-keygen -t rsa -b 4096 -C "k4kranti@gmail.com" -f ~/.ssh/id_rsa_kranti

#ensure that the SSH agent is running and add ssh key 

eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_rsa_kranti

# Add SSH keys to Git account

#copy key to clipboard and add in your git account --> settings -> SSH keys

cat ~/.ssh/id_rsa_kranti.pub | pbcopy
#https://github.com/settings/keys

# Now config your git config

#Create git config file ~/.ssh/config 

 touch config

# Personal Kranti GitHub account
Host github.com-kranti
	HostName github.com
	User git
	IdentityFile ~/.ssh/id_rsa_kranti

 git clone git@github.com:k4kranti/cheatsheets.git
